function renderTraining(){
 const grid=document.getElementById('days-grid');
 grid.innerHTML=DAYS.map(d=>{
   const macros=S.phase==='cut'?d.cutMacros:d.massMacros;
   const exes=d.sessions.flatMap(s=>s.exercises);
   const total=exes.length;const done=exes.filter(e=>S.exDone[d.id+'_'+e.name]).length;
   const pct=total>0?Math.round(done/total*100):0;
